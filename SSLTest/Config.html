def loglevel
  levelstr = if @opts.has_key? :loglevel
    @opts[:loglevel].upcase
  elsif @raw.has_key? 'log' and @raw['log'].has_key? 'level'
    @raw['log']['level'].upcase
  else
    'INFO'
  end
  Logger.const_get(levelstr)
end

def packetthief
  pt = @raw['packetthief'].dup
  newvals = {}
  pt.each_pair do |k,v|
    if k.kind_of? String
      newvals[k.to_sym] = v
    end
  end
  pt.merge! newvals
  pt
end
