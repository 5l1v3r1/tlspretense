def client_connected
  @orig_dest_host = dest_host
  @orig_dest_port = dest_port
  super
  send_to_dest("CONNECT #{@orig_dest_host}:#{@orig_dest_port} HTTP/1.1\r\n\r\n")
  @proxy_state = :sentconnect
end

def client_recv(data)
  case @proxy_state
  when :init, :sentconnect
    @proxy_buf << data
  when :proxying
    if @proxy_buf
      send_to_dest @proxy_buf if @proxy_buf and @proxy_buf != ''
      @proxy_buf = nil
    end
    send_to_client data
  end
end

def dest_recv(data)
  case @proxy_state
  when :sentconnect
    logdebug "Received response to CONNECT", :response => data
    successfulre = %r\A(?<headers>HTTP\/1\..? 200.*?\r\n\r\n)(?<therest>.*)\z/
    m = successfulre.match(data)
    if m
      @proxy_state = :proxying
      send_to_dest(@proxy_buf) if @proxy_buf and @proxy_buf != ''
      @proxy_buf = nil
      send_to_client(m[:therest])
    else
      raise "response did not match: #{data.inspect}"
    end
  when :proxying
    send_to_client(data)
  else
    raise "unhandled proxy state: #{@proxy_state}"
  end
end
